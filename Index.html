          {evts.map(ev=><MobCard key={ev.id} ev={ev} isExp={expCard===ev.id} onClick={()=>{if(expCard!==ev.id)setExpCard(ev.id);else{setSelId(ev.id);setDet(true);}}}/>)}

        </div>)}

      </div>

        </div>

      </div>

          {lr.badges.map(b=>{

            const bt=AY+CG+ml*(cH+12)+cH+16;let bx=b.cx;for(let y=SY;y<=EY;y++){if(Math.abs(bx-gp(y,1,ppY))<40){bx=gp(y,1,ppY)+50;break;}}

      <Start vis={start} onGo={()=>setStart(false)} mob={false}/>

      {det&&!start&&<Detail ev={selEv} onNext={nxt} onPrev={prv} onClose={()=>setDet(false)}/>}

  );
